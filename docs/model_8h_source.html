32 /*
33  * set of macros to help generate struct and function for our model;
34  *
35  * - DECLARE_MODEL(type, model_def) :
36  * `type` name of the struct,
37  * `model_def` - marco defining fields for the model
38  *
39  * NOTE: matching MODEL_IMPL macro in model.c is used to generate function implementations
40  *
41  * Fields are defined with name, type, type modifier, and path (for JSON mapping)
42  *
43  * The following functions are generated:
44  * - TYPE* parse_TYPE(json, len) -- parses json into an allocated struct
45  *
46  * - void free_TYPE(TYPE *obj) -- frees struct
47  *
48  * - int dump_TYPE(TYPE *obj, int indent) -- prints the struct to `stdout`,
49  * `indent` is used for printing nested model objects
50  */
51 
52 #define FIELD_DECL(name, type, mod, path) mod(type) name;
53 
54 /*
55  * field type macros
56  *
57  * Nested model objects are supported as pointers and NULL-terminated arrays of pointers
58  */
59 #define none(type) type
60 #define array(type) type**
61 #define ptr(type) type*
62 
63 /*
64  * Model declaration: struct type, and functions
65  */
66 #define DECLARE_MODEL(type, model) \
67 typedef struct type##_s {\
68 model(FIELD_DECL) \
69 SLIST_ENTRY(type##_s) _next; \
70 } type;\
71 typedef SLIST_HEAD(type##_l, type##_s) type##_list;\
72 type* parse_##type(const char* json, int json_len);\
73 type** parse_##type##_array(const char* json, int json_len);\
74 void free_##type(type* type);\
75 void free_##type##_array(type** arr); \
76 void free_##type##_list(type##_list* l); \
77 int dump_##type(type* type, int len);
